### 安装
```
#vue-cli
cnpm install -g vue-cli
#初始化项目
vue init webpack x
#运行
cd x
npm run dev
#打包
npm run build
```
### 其他
```
#安装插件
cnpm install vue-router --save
cnpm install vuex --save

cnpm install vue-wechat-title --save
```
